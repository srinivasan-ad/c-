void Print(vector<int>& arr)
{
    for(int i = 0  ; i < arr.size() ; i++)
    {
        cout << arr[i] << " ";
    }
}

void Print(vector<vector<int>>& mat)
{
    for(int i = 0  ; i < mat.size() ; i++)
    {
        for(int j = 0 ; j  <mat[0].size() ; j++)
        {

            cout << mat[i][j] << " ";
        }
        cout << endl;
    }
}
